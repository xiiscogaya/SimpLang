def int suma(int a, int b) {
    return a + b;
}

def int cuadradoDeSuma(int a, int b) {
    int resultado;
    resultado = suma(a, b);
    resultado = suma(resultado, resultado);
    return resultado;
}

def bool esPositivo(int n) {
    if n > 0 {
        return true;
    } 
    return false;
}

def int factorial(int n) {
    int resultado;
    resultado = 1;
    while n > 0 {
        resultado = resultado * n;
        n = n - 1;
    }
    return resultado;
}

{
    # Uso de una función con variables

    int counter;
