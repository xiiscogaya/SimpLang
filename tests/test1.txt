# PROGRAMA 1 - FUNCIONA
# Realiza el print del numero pedido 10 veces, si el valor es mayor a 5, imprime el valor
# sino imprime el valor 0

{
    int count;
    count = 0;

    while count < 10 {
        int a;
        input(a);
        if a > 5 {
            print(a);
        } else {
            print(0);
        }
        count = count + 1;
    }
}
