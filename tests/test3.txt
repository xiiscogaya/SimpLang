# Programa que inicializa un array del 1 al 5, y despues va multiplicando sus números con una función auxiliar
# cuando la multiplicacion supere el valor de 10, imprimirá el valor total calculado hasta el momento

def int mult(int a, int b) {
    return a * b;
}

{
    # Declaramos el array con números
    array int nums[5];
    nums[0] = 1;
    nums[1] = 2;
    nums[2] = 3;
    nums[3] = 4;
    nums[4] = 5;


    int total;
    total = 1;

    bool i;
    i = true;

    int j;
    j = 0;

    while i {
        total = mult(total, nums[j]);
        if total > 10 {
            print(total);
            i = false;
        }
        j = j + 1;
    }
}
